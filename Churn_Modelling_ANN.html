def one_hot_encode(y):
  N=len(y)
  K=len(set(y))
  Y=np.zeros((N,K))
  for i in range(N):
    Y[i,y[i]]=1
  return Y

def binary_cross_entropy(Y, P_hat):
  epsilon = 1e-9
  return -np.mean(Y*np.log(P_hat + epsilon) + (1 -Y) * np.log(1 - P_hat + epsilon))

def accuracy(y, y_hat):
  return np.mean(y==y_hat)

class Shallow_ANN():

  def fit(self, X, y, neurons=6, eta=1e-3, epochs=1e3, mode='linear', show_curve=True):

    epochs = int(epochs)
    N, D = X.shape
    Y = y.reshape(-1, 1)
    K = Y.shape[1]
    # Initilize Weights and Biases
    self.W = {l: np.random.randn(M[0],M[1]) * 0.01 for l, M in enumerate(zip([D,neurons],[neurons,K]), 1)}
    self.B = {l: np.random.randn(M) * 0.01 for l, M in enumerate([neurons, K],1)}
    # Define Activations
    self.a = {1:np.tanh, 2:Sigmoid}

    J = np.zeros(epochs)

    #SGD Steps
    for epoch in range(epochs):
      self.__foward__(X)
      J[epoch] = binary_cross_entropy(Y, self.Z[2])

      #Weight Update Rules for Layer 2
      self.W[2] -= eta*(1/N)*self.Z[1].T@(self.Z[2]-Y)
      self.B[2] -= eta*(1/N)*(self.Z[2]-Y).sum(axis=0)
      #Weight Update for Layer 1
      self.W[1] -= eta*(1/N)*X.T@((self.Z[2]-Y)@self.W[2].T*(1-self.Z[1]**2))
      self.B[1] -= eta*(1/N)*((self.Z[2]-Y)@self.W[2].T*(1-self.Z[1]**2)).sum(axis=0)

    if show_curve:
      plt.figure()
      plt.plot(J)
      plt.xlabel("epochs")
      plt.ylabel("$\mathcal{J}$")
      plt.title("Training Curve")

  def __foward__(self, X):
      self.Z ={0:X}
      for l in sorted(self.W.keys()):
        self.Z[l] = self.a[l](self.Z[l-1]@self.W[l]+self.B[l])
  def predict(self, X):
      self.__foward__(X)
      return (self.Z[2] > 0.5).astype(int).flatten()
